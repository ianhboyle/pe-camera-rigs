### Property Group Definition

```python
# File: addons/vr_production_toolkit/ui/properties.py

import bpy
import math
from bpy.props import (
    FloatProperty,
    IntProperty,
    EnumProperty,
    BoolProperty,
    StringProperty,
)
from bpy.types import PropertyGroup


class IsometricSettings(PropertyGroup):
    """Settings for isometric orthographic camera"""

    # ==========================================
    # Projection Type
    # ==========================================

    projection_type: EnumProperty(
        name="Projection Type",
        description="Type of parallel projection",
        items=[
            # Game & Pixel Art (Most Common)
            ('GAME_ISOMETRIC', "Game Isometric (2:1)", "26.565° tilt • Perfect for strategy games • 2:1 pixel ratio stairs"),
            ('GAME_4TO3', "Game 4:3 Ratio", "30° tilt • Perfect for RPG tiles • Diamond tiles in 4:3 aspect"),

            # Technical & CAD
            ('TRUE_ISOMETRIC', "True Isometric", "35.264° tilt • Mathematically correct • All axes equally foreshortened"),
            ('DIMETRIC', "Dimetric (30°)", "30° tilt • Two axes equal • Common in Japanese games"),
            ('TRIMETRIC', "Trimetric", "60° + 30° • All axes different • Unique perspectives"),

            # Plans & Maps
            ('MILITARY', "Military (Top-Down)", "90° tilt • Perfect top-down view • Floor plans, RTS games"),
            ('CAVALIER', "Cavalier", "0° + 45° roll • Front face true size • Cabinet drawings"),

            # Custom
            ('CUSTOM', "Custom Angles", "User-defined rotation and tilt"),
        ],
        default='TRUE_ISOMETRIC',  # True isometric as default (most mathematically correct)
    )

    # ==========================================
    # Camera Angles
    # ==========================================

    rotation_z: FloatProperty(
        name="Rotation (Z)",
        description="Horizontal rotation around Z-axis (pan angle)",
        default=math.radians(45.0),
        min=math.radians(-180.0),
        max=math.radians(180.0),
        step=100,
        precision=2,
        subtype='ANGLE',
    )

    tilt_x: FloatProperty(
        name="Tilt (X)",
        description="Vertical rotation around X-axis (pitch angle)",
        default=math.atan(1.0 / math.sqrt(2.0)),  # 35.264° for true isometric
        min=math.radians(-90.0),
        max=math.radians(90.0),
        step=100,
        precision=2,
        subtype='ANGLE',
    )

    roll_y: FloatProperty(
        name="Roll (Y)",
        description="Camera roll around Y-axis (usually 0° for isometric)",
        default=0.0,
        min=math.radians(-180.0),
        max=math.radians(180.0),
        step=100,
        precision=2,
        subtype='ANGLE',
    )

    # ==========================================
    # Orthographic Scale
    # ==========================================

    ortho_scale: FloatProperty(
        name="Orthographic Scale",
        description="Orthographic camera scale (zoom level)",
        default=14.0,  # IsoCam standard
        min=0.1,
        max=1000.0,
        soft_min=1.0,
        soft_max=100.0,
        step=10,
        precision=2,
    )

    # ==========================================
    # Camera Position
    # ==========================================

    camera_distance: FloatProperty(
        name="Camera Distance",
        description="Distance from origin (target point)",
        default=15.0,
        min=1.0,
        max=1000.0,
        soft_min=5.0,
        soft_max=50.0,
        unit='LENGTH',
        precision=2,
        subtype='DISTANCE',
    )

    camera_height: FloatProperty(
        name="Camera Height",
        description="Camera Z height above ground plane",
        default=10.0,
        min=-100.0,
        max=100.0,
        soft_min=0.0,
        soft_max=30.0,
        unit='LENGTH',
        precision=2,
        subtype='DISTANCE',
    )

    # ==========================================
    # Visual Aids
    # ==========================================

    show_grid: BoolProperty(
        name="Show Grid",
        description="Display reference grid plane",
        default=True,
    )

    show_axes: BoolProperty(
        name="Show Axes",
        description="Display X, Y, Z axis indicators",
        default=False,
    )

    show_scale_reference: BoolProperty(
        name="Show Scale Reference",
        description="Display measurement reference objects",
        default=False,
    )

    grid_size: FloatProperty(
        name="Grid Size",
        description="Size of reference grid",
        default=20.0,
        min=1.0,
        max=1000.0,
        unit='LENGTH',
        precision=1,
        subtype='DISTANCE',
    )

    # ==========================================
    # Resolution Presets
    # ==========================================

    resolution_preset: EnumProperty(
        name="Resolution Preset",
        description="Common resolution presets for isometric rendering",
        items=[
            ('1K_SQUARE', "1K Square", "1024×1024 (game assets, icons)"),
            ('2K_SQUARE', "2K Square", "2048×2048 (high-quality icons)"),
            ('4K_SQUARE', "4K Square", "4096×4096 (print-ready)"),
            ('HD', "HD 1080p", "1920×1080 (presentation)"),
            ('4K', "4K 2160p", "3840×2160 (high-res documentation)"),
            ('CUSTOM', "Custom", "User-defined resolution"),
        ],
        default='2K_SQUARE',
    )

    resolution_x: IntProperty(
        name="Resolution X",
        description="Render width in pixels",
        default=2048,
        min=128,
        max=16384,
        subtype='PIXEL',
    )

    resolution_y: IntProperty(
        name="Resolution Y",
        description="Render height in pixels",
        default=2048,
        min=128,
        max=16384,
        subtype='PIXEL',
    )

    # ==========================================
    # Render Settings
    # ==========================================

    render_quality: EnumProperty(
        name="Render Quality",
        description="Quality preset for rendering",
        items=[
            ('DRAFT', "Draft", "64 samples, fast preview"),
            ('PREVIEW', "Preview", "128 samples, balanced"),
            ('FINAL', "Final", "256 samples, high quality"),
            ('ULTRA', "Ultra", "512 samples, maximum quality"),
            ('CUSTOM', "Custom", "User-defined samples"),
        ],
        default='PREVIEW',
    )

    render_samples: IntProperty(
        name="Render Samples",
        description="Number of render samples",
        default=128,
        min=1,
        max=4096,
    )

    use_transparent: BoolProperty(
        name="Transparent Background",
        description="Render with transparent alpha channel",
        default=True,
    )

    use_freestyle: BoolProperty(
        name="Freestyle Lines",
        description="Enable Freestyle line rendering for technical drawings",
        default=False,
    )

    freestyle_thickness: FloatProperty(
        name="Line Thickness",
        description="Freestyle line thickness in pixels",
        default=1.5,
        min=0.1,
        max=10.0,
        precision=1,
    )

    # ==========================================
    # Scene Setup Options
    # ==========================================

    add_lighting: BoolProperty(
        name="Add Lighting",
        description="Add default three-point lighting setup",
        default=True,
    )

    lighting_preset: EnumProperty(
        name="Lighting Preset",
        description="Lighting setup style",
        items=[
            ('TECHNICAL', "Technical", "Flat, shadowless lighting for technical drawings"),
            ('SOFT', "Soft", "Subtle directional lighting with soft shadows"),
            ('STUDIO', "Studio", "Three-point lighting for presentation"),
            ('GAME', "Game", "Directional from top-left for game assets"),
        ],
        default='SOFT',
    )

    add_reference_objects: BoolProperty(
        name="Add Reference Objects",
        description="Add scale reference objects to scene",
        default=False,
    )
```

def create_isometric_camera(
    context: bpy.types.Context,
    name: str = "IsometricCamera",
    projection_type: str = 'TRUE_ISOMETRIC',
    ortho_scale: float = 14.0,
    location: tuple[float, float, float] = (0.0, 0.0, 0.0),
) -> tuple[bpy.types.Object, bpy.types.Object]:
    """
    Create isometric orthographic camera with parent empty.

    Args:
        context: Blender context
        name: Base name for camera and parent empty
        projection_type: Preset projection type
            - 'GAME_ISOMETRIC': Game Isometric (2:1)
            - 'GAME_4TO3': Game 4:3 Ratio
            - 'TRUE_ISOMETRIC': True Isometric (default)
            - 'DIMETRIC': Dimetric (30°)
            - 'TRIMETRIC': Trimetric
            - 'MILITARY': Military (Top-Down)
            - 'CAVALIER': Cavalier
            - 'CUSTOM': Custom angles
        ortho_scale: Orthographic camera scale (14.0 = IsoCam standard)
        location: World location for camera parent

    Returns:
        Tuple of (parent_empty, camera_object)

    Creates:
        - Parent empty at location
        - Orthographic camera as child of empty
        - Sets perfect isometric angles for preset
        - Adds to 'Isometric_Cameras' collection
        - Auto-activates camera view (IsoCam-style)
    """
    pass


def set_isometric_angles(
    camera: bpy.types.Object,
    projection_type: str = 'ISOMETRIC',
    rotation_z: float = None,
    tilt_x: float = None,
    roll_y: float = None,
) -> tuple[float, float, float]:
    """
    Set camera rotation to isometric or custom angles.

    Args:
        camera: Camera object to rotate
        projection_type: Preset projection type
        rotation_z: Override Z rotation (radians), None uses preset
        tilt_x: Override X tilt (radians), None uses preset
        roll_y: Override Y roll (radians), None uses preset

    Returns:
        Tuple of (rotation_z, tilt_x, roll_y) in radians

    Projection presets:
        GAME_ISOMETRIC: (45°, 26.565°, 0°) - 2:1 pixel ratio stairs
        GAME_4TO3: (45°, 30°, 0°) - 4:3 diamond tiles
        TRUE_ISOMETRIC: (45°, 35.264°, 0°) - Mathematically correct
        DIMETRIC: (45°, 30°, 0°) - Two axes equal
        TRIMETRIC: (60°, 30°, 0°) - All axes different
        MILITARY: (0° or 45°, 90°, 0°) - Top-down view
        CAVALIER: (0°, 0°, 45°) - Front face true size
    """
    pass

# File: addons/vr_production_toolkit/__init__.py

from bpy.utils import register_class, unregister_class
from .ui.properties import IsometricSettings

def register():
    register_class(IsometricSettings)
    bpy.types.Scene.isometric_settings = bpy.props.PointerProperty(
        type=IsometricSettings
    )

def unregister():
    del bpy.types.Scene.isometric_settings
    unregister_class(IsometricSettings)
